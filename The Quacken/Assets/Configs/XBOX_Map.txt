Buttons
    A		= joystick button 0
    B       	= joystick button 1
    X  		= joystick button 2
    Y 		= joystick button 3
    L-Bumper    = joystick button 4
    R-Bumper	= joystick button 5
    Back	= joystick button 6
    Start 	= joystick button 7
    L-Stick     = joystick button 8
    R-Stick     = joystick button 9

Axes:
    LeftStickX  = X-Axis
    LeftStickY	= Y-Axis
    RightStickX	= 4rd Axis
    RightStickY	= 5th Axis
    L2        	= 9th Axis
    R2        	= 10th Axis
    DPadX     	= 6th Axis
    DPadY     	= 7th Axis
